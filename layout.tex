\newcommand{\mygrid}[5]{%
  \pgfmathsetmacro{\xmin}{#1+1}
  \pgfmathsetmacro{\xmax}{#1+#3-1}
  \pgfmathsetmacro{\ymin}{#2+1}
  \pgfmathsetmacro{\ymax}{#2+#4-1}
  \ifthenelse{\equal{#5}{}}
  {\draw (#1,#2) rectangle ++(#3,#4);}
  {\draw[fill=#5] (#1,#2) rectangle ++(#3,#4);}
  \foreach \x in {\xmin,...,\xmax}{
    \draw (\x,#2)-- ++(0,#4);
  }
  \foreach \y in {\ymin,...,\ymax}{
    \draw (#1,\y)-- ++(#3,0);
  }
}
